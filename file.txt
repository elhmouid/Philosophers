#include "philosophers.h"

int ft_atoi(char *str)
{
    long	result;
	int		sign;
	int		i;

	result = 0;
	sign = 1;
	i = 0;
	if (!str[i])
		return (0);
	while (str[i] == ' ' || (str[i] >= 9 && str[i] <= 13))
		i++;
	if (str[i] == '-' || str[i] == '+')
	{
		if (str[i] == '-')
			sign = -1;
		i++;
	}
	while (str[i] >= '0' && str[i] <= '9')
	{
		result = (result * 10) + (str[i] - '0');
		i++;
	}
	return (result * sign);
}

int	ft_isdigit(int c)
{
	return (c >= '0' && c <= '9');
}

int	parsing_arguments(char *av)
{
	int	i;

	i = 0;
	while (av[i])
	{
		if (!ft_isdigit(av[i]))
			return (1);
		i++;
	}
	return (0);
}

void	fill_the_struct(int ac, char **av, t_philo *philo)
{
		philo->number_of_philosophers = ft_atoi(av[1]);
		philo->time_to_die = ft_atoi(av[2]);
		philo->time_to_eat = ft_atoi(av[3]);
		philo->time_to_sleep = ft_atoi(av[4]);
		if (ac == 6)
			philo->number_of_times_each_philosopher_must_eat = ft_atoi(av[5]);
}/* ************************************************************************** */

int main(int ac, char **av)
{
	t_philo *philo = malloc(sizeof(t_philo));
	int	i = 0;
	if (ac <= 4 || ac > 6)
	{
		printf("Error in the Arguments");
		return (0);
	}
	while (i++ < ac - 1)
	{
		if (parsing_arguments(av[i]) == 1)
		{
			printf("Error in the Arguments");
			return (1);
		}
	}
		fill_the_struct(ac, av, philo);
		printf("number of philo is : %d\n", philo->number_of_philosophers);
		printf("time to die is : %d\n", philo->time_to_die);
		printf("time to eat is : %d\n", philo->time_to_eat);
		printf("time to slepp is : %d\n", philo->time_to_sleep);

}
